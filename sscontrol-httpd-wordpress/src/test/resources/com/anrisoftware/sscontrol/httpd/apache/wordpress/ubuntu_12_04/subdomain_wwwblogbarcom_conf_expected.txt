    # domain www.blogbar.com
    ServerName www.blogbar.com
    DocumentRoot /tmp/var/www/www.blogbar.com/web
    SuexecUserGroup web_001 web_001
    <Directory /tmp/var/www/www.blogbar.com/web>
        AllowOverride None
        Order allow,deny
        Allow from all
    </Directory>

    # /tmp/var/www/www.test1.com/wordpress_4

    Alias / /tmp/var/www/www.test1.com/wordpress_4/

    <Directory /tmp/var/www/www.test1.com/wordpress_4>
        AddHandler fcgid-script .php
        FCGIWrapper /tmp/var/www/php-fcgi-scripts/www.blogbar.com/php-fcgi-starter .php
        Options +ExecCGI +FollowSymLinks
        DirectoryIndex index.php
        AllowOverride All

        Order allow,deny
        Allow from all

        <IfModule mod_rewrite.c>
            # block comment spam by denying access to no-referrer requests
            RewriteEngine On
            RewriteCond %{REQUEST_METHOD} POST
            RewriteCond %{REQUEST_URI} .wp-comments-post\.php*
            RewriteCond %{HTTP_REFERER} !.*www\.blogbar\.com.* [OR]
            RewriteCond %{HTTP_USER_AGENT} ^$
            RewriteRule ^(.*)$ ^http://%{REMOTE_ADDR}/$ [R=301,L]
        </IfModule>

        <IfModule mod_mime.c>
            <IfModule !mod_php5.c>
                <IfModule mod_actions.c>
                    <IfModule mod_cgi.c>
                        AddType application/x-httpd-php .php
                        Action application/x-httpd-php /cgi-bin/php5
                    </IfModule>
                    <IfModule mod_cgid.c>
                        AddType application/x-httpd-php .php
                        Action application/x-httpd-php /cgi-bin/php5
                    </IfModule>
                </IfModule>
            </IfModule>
        </IfModule>
    </Directory>
