<VirtualHost 192.168.0.50:80>

    # domain('anrisoftware.com')
    ServerName anrisoftware.com
    DocumentRoot /var/www/anrisoftware.com/web
    <Directory /var/www/anrisoftware.com/web>
        AllowOverride None
        Order allow,deny
        Allow from all
    </Directory>
    
    # redirect_to_www
    ServerAlias www.anrisoftware.com
    RewriteEngine On
    RewriteCond %{HTTP_HOST} ^anrisoftware\.com$ [NC]
    RewriteRule ^(.*)$ http://www.anrisoftware.com$1 [L,R=301]
    
    # redirect_http_to_https
    RewriteEngine On
    RewriteCond %{HTTP_HOST} ^anrisoftware.com
    RewriteRule (.*) https://anrisoftware.com$1 [R=301,L]
    
    Alias / /var/www/anrisoftware.com/web/
    
    # end

</VirtualHost>
