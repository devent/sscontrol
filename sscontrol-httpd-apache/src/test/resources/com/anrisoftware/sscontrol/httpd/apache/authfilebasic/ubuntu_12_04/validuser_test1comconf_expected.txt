<VirtualHost 192.168.0.50:80>

    ServerName test1.com
    DocumentRoot /tmp/var/www/test1.com/web
    SuexecUserGroup web_001 web_001
    <Directory /tmp/var/www/test1.com/web>

        AuthBasicProvider file
        Require valid-user

        Satisfy any
    </Location>
