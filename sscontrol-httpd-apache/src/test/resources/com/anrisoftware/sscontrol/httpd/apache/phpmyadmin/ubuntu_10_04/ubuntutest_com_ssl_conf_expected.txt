<VirtualHost 192.168.0.100:443>

    # domain ubuntutest.com
    ServerName ubuntutest.com
    DocumentRoot /tmp/var/www/ubuntutest.com/web
    SuexecUserGroup web_001 web_001
    <Directory /tmp/var/www/ubuntutest.com/web>
        AllowOverride None
        Order allow,deny
        Allow from all
    </Directory>

    SSLEngine on
    SSLCertificateFile /tmp/var/www/ubuntutest.com/ssl/cert_crt.txt
    SSLCertificateKeyFile /tmp/var/www/ubuntutest.com/ssl/cert_key.txt

    # phpmyadmin

    Alias /phpmyadmin /usr/share/phpmyadmin

    <Directory /usr/share/phpmyadmin>
        AddHandler fcgid-script .php
        FCGIWrapper /tmp/var/www/php-fcgi-scripts/ubuntutest.com/php-fcgi-starter .php
        Options +ExecCGI
        DirectoryIndex index.php
    </Directory>

    # Disallow web access to directories that don't need it
    <Directory /usr/share/phpmyadmin/libraries>
        Order Deny,Allow
        Deny from All
    </Directory>
    <Directory /usr/share/phpmyadmin/setup/lib>
        Order Deny,Allow
        Deny from All
    </Directory>

</VirtualHost>
